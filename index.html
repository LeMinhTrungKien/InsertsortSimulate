	def insertion_sort(arr): 
	n = len(arr)
    	for i in range(1, n):
       		key = arr[i]  # phần tử hiện tại cần chèn
       		j = i - 1  # Vị trí của phần tử trước đó
     		# Di chuyển các phần tử lớn hơn key về phía sau
    		while j >= 0 and arr[j] > key:
    			arr[j + 1] = arr[j]
			j -= 1
			arr[j + 1] = key  # Chèn key vào vị trí đúng
			i=i+1
